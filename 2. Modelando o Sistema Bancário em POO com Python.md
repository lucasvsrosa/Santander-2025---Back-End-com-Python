```Python
import re
import csv

# Fun√ß√£o para validar CPF
def validar_cpf(cpf):
    cpf = re.sub(r'[^0-9]', '', cpf)  # Remover caracteres n√£o num√©ricos
    if len(cpf) != 11:
        return False
    # Valida√ß√£o simples (n√£o implementa o c√°lculo completo, apenas formato)
    return True

# Classe Conta
class Conta:
    def __init__(self, titular, saldo=0):
        self.titular = titular
        self.saldo = saldo
        self.transacoes = []

    def registrar_transacao(self, tipo, valor):
        self.transacoes.append((tipo, valor))

    def consultar_historico(self):
        print(f"Hist√≥rico de transa√ß√µes de {self.titular}:")
        for transacao in self.transacoes:
            print(f"{transacao[0]} de R${transacao[1]:.2f}")

    def depositar(self, valor):
        if valor > 0:
            self.saldo += valor
            self.registrar_transacao("Dep√≥sito", valor)
            print(f"Dep√≥sito de R${valor:.2f} realizado com sucesso!")

    def sacar(self, valor):
        if valor > 0 and valor <= self.saldo:
            self.saldo -= valor
            self.registrar_transacao("Saque", valor)
            print(f"Saque de R${valor:.2f} realizado com sucesso!")
        else:
            print("Saldo insuficiente ou valor inv√°lido para saque!")

    def transferir(self, valor, conta_destino):
        if valor > 0 and valor <= self.saldo:
            self.saldo -= valor
            conta_destino.depositar(valor)
            self.registrar_transacao("Transfer√™ncia", valor)
            print(f"Transfer√™ncia de R${valor:.2f} realizada com sucesso!")
        else:
            print("Saldo insuficiente ou valor inv√°lido para transfer√™ncia!")

    def consultar_saldo(self):
        return f"Saldo atual de {self.titular}: R${self.saldo:.2f}"


# Classe Conta Corrente (com cheque especial)
class ContaCorrente(Conta):
    def __init__(self, titular, saldo=0, limite_cheque=500):
        super().__init__(titular, saldo)
        self.limite_cheque = limite_cheque

    def sacar(self, valor):
        if valor > 0 and valor <= (self.saldo + self.limite_cheque):
            self.saldo -= valor
            self.registrar_transacao("Saque", valor)
            print(f"Saque de R${valor:.2f} realizado com sucesso!")
        else:
            print("Saldo insuficiente, incluindo o limite do cheque especial!")


# Classe Conta Poupan√ßa (com juros)
class ContaPoupanca(Conta):
    def __init__(self, titular, saldo=0, taxa_juros=0.05):
        super().__init__(titular, saldo)
        self.taxa_juros = taxa_juros

    def aplicar_juros(self):
        juros = self.saldo * self.taxa_juros
        self.saldo += juros
        self.registrar_transacao("Juros", juros)
        print(f"Juros de {self.taxa_juros * 100}% aplicados! Novo saldo: R${self.saldo:.2f}")


# Classe Cliente
class Cliente:
    def __init__(self, nome, cpf):
        if validar_cpf(cpf):
            self.nome = nome
            self.cpf = cpf
            self.contas = []
            print(f"Cliente {self.nome} com CPF {self.cpf} cadastrado com sucesso!")
        else:
            print("CPF inv√°lido!")

    def adicionar_conta(self, conta):
        self.contas.append(conta)
        print(f"Conta para {self.nome} criada com sucesso!")

    def consultar_contas(self):
        print(f"Contas de {self.nome}:")
        for conta in self.contas:
            print(f"{conta.consultar_saldo()}")


# Classe Banco
class Banco:
    def __init__(self, nome):
        self.nome = nome
        self.clientes = []

    def adicionar_cliente(self, cliente):
        self.clientes.append(cliente)
        print(f"Cliente {cliente.nome} adicionado ao banco {self.nome}!")

    def listar_clientes(self):
        print(f"Lista de clientes do banco {self.nome}:")
        for cliente in self.clientes:
            print(f"Nome: {cliente.nome}, CPF: {cliente.cpf}")

    def consultar_cliente(self, cpf):
        for cliente in self.clientes:
            if cliente.cpf == cpf:
                print(f"Cliente encontrado: {cliente.nome}, CPF: {cliente.cpf}")
                cliente.consultar_contas()
                return
        print("Cliente n√£o encontrado!")

    def remover_cliente(self, cpf):
        for cliente in self.clientes:
            if cliente.cpf == cpf:
                self.clientes.remove(cliente)
                print(f"Cliente {cliente.nome} removido do banco {self.nome}!")
                return
        print("Cliente n√£o encontrado!")

    def transferir_entre_bancos(self, conta_origem, conta_destino, valor):
        if conta_origem.saldo >= valor:
            conta_origem.saldo -= valor
            conta_destino.depositar(valor)
            print(f"Transfer√™ncia de R${valor:.2f} entre bancos realizada com sucesso!")
        else:
            print("Saldo insuficiente para realizar a transfer√™ncia!")

    def gerar_relatorio(self):
        with open("relatorio_bancario.csv", mode='w', newline='') as file:
            writer = csv.writer(file)
            writer.writerow(["Cliente", "Saldo"])
            for cliente in self.clientes:
                for conta in cliente.contas:
                    writer.writerow([cliente.nome, conta.saldo])
        print("Relat√≥rio gerado com sucesso!")


# Exemplo de uso
banco1 = Banco("Banco Digital")

# Criando clientes
cliente1 = Cliente("Jo√£o Silva", "123.456.789-00")
cliente2 = Cliente("Maria Oliveira", "987.654.321-00")

# Criando contas para os clientes
conta1 = ContaCorrente(cliente1.nome, 2000)
conta2 = ContaPoupanca(cliente2.nome, 1000)

# Adicionando contas aos clientes
cliente1.adicionar_conta(conta1)
cliente2.adicionar_conta(conta2)

# Adicionando clientes ao banco
banco1.adicionar_cliente(cliente1)
banco1.adicionar_cliente(cliente2)

# Listando clientes do banco
banco1.listar_clientes()

# Consultando conta de um cliente
banco1.consultar_cliente("123.456.789-00")

# Transferindo entre contas de clientes do mesmo banco
conta1.transferir(500, conta2)

# Aplicando juros na conta poupan√ßa
conta2.aplicar_juros()

# Gerando relat√≥rio banc√°rio
banco1.gerar_relatorio()

# Removendo cliente do banco
banco1.remover_cliente("987.654.321-00")
```
